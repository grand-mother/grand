    def test_init_traces(self):
        assert self.tr3d.traces[0, 0, 0] == 0.0
        assert self.tr3d.traces[2, 2, 3] == self.size_tot - 1
        assert self.tr3d.du_id[-1] == 2
        assert np.allclose(self.tr3d.t_start_ns, np.array([100, 110, 120]))
